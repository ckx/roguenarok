function	script	ChangeJob	{
	.@jobToBe$ = getarg(0);
	dispbottom "You will become a " + .@jobToBe$;

	if	(.@jobToBe$ == "Swordman") {
		jobchange Job_Swordman;
	}
	
	close;
}
